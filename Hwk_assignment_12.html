1.Create four random undirected Barabasi-Albert graphs with 5000 nodes each with p=1, m=3.

Creation of 4 random graphs:

g <- lapply(rep(5000,4), function(node) {
  barabasi.game(node, p=1, m=3, directed=FALSE)
})
a.Calculate the average path lengths for each of the graphs. Display the results and the execution times.
start_time <- Sys.time()
results <- foreach(iter(1:4), .combine='cbind') %do% average.path.length(barabasi.game(5000, p=1, m=3, directed=FALSE))
results
##      result.1 result.2 result.3 result.4
## [1,]        4        4        4        4
sequential.apl <- Sys.time() - start_time
sequential.apl
## Time difference of 11 secs
b.Implement (a) in parallel.
# Let's find the number of cores in my laptop
numParCores <- max(1, detectCores()-1)
# Let's make clusters
cl <- makeCluster(numParCores)
# Let's register clusters
registerDoParallel(cl)
start_time <- Sys.time()
results <- foreach(icount(4), .combine="cbind", .packages="igraph") %dopar% { average.path.length(barabasi.game(5000, p=1, m=3, directed=FALSE))
}
stopCluster(cl)
results
##      result.1 result.2 result.3 result.4
## [1,]        4        4        4        4
parallel.apl <- Sys.time() - start_time
parallel.apl
## Time difference of 6 secs
